#include
int main(){
printf("Enter the number of processes\n");
int n;
scanf("%d",&n);
int Arrival_time[n];
int Burst_time[n];
int Completion_time[n];
int TAT[n];
int WT[n];
for(int i=0;i printf("Enter the arrival time of %d process\n",i+1);
scanf("%d",&Arrival_time[i]);
}
for(int i=0;i printf("Enter the Burst time of %d process\n",i+1);
scanf("%d",&Burst_time[i]);
}
Completion_time[0]=Burst_time[0];
for(int i=1;i // printf("Hello World");
if(Arrival_time[i]<=Completion_time[i-1]){
Completion_time[i]=Completion_time[i-1]+Burst_time[i];
}
else{
Completion_time[i]=Arrival_time[i]+Burst_time[i];
}
}
for(int i=0;i // printf("Hello World");
TAT[i]=Completion_time[i]-Arrival_time[i];
}
for(int i=0;i // printf("Hello World");
WT[i]=TAT[i]-Burst_time[i];
}
char l[1]="P";
printf("This is the outcome :\n");
char a[20]="Process NO:";
char b[20]="Arrival time";
char c[20]="Burst time";
char d[20]="Completion time";
char e[20]="Turn around time";
char f[20]="Waiting time";
printf("%-20s %-20s %-20s %-20s %-20s %-20s\n",a,b,c,d,e,f);
for(int i=0;i printf("%s%-19d %-20d %-20d %-20d %-20d %-20d\n",l,i+1,Arrival_time[i],Burst_time[i],Completion_time[i],TAT[i],WT[i]);
}

return 0;
}
