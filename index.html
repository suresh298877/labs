knowledge_table={
('dirty','A'):'Suck',
('dirty','B'):'Suck',
("dirty",'C'):'Suck',
('clean','A'):'right',
('clean','B'):'down',
('clean','C'):'left'
}

def agent(location,state):
action=knowledge_table.get((state,location),'invalid')
if action=='suck':
print('Agent cleaned',location)
elif action=='right':
print("Agent moved right from",location)
location='B'
elif action=='down':
print("Agent moved down from",location)
location='C'
elif action=='left':
print('Agent moved left from',location)
location='B'
else:
print('Invalid state-action pair:',state,location)
return location

environment={
'A':'dirty',
'B':'dirty',
'C':'dirty'
}

def main():
location='A'
while True:
state=environment[location]
agent(location,state)
environment[location]='clean'
if all(value == 'clean' for value in environment.values()):
print("Environment is clean!")
break
location=agent(location,state)
if __name__=='__main__':
main()







def reflex_agent(location,status):
if status=='dirty':
return 'suck'
elif location=='A':
return 'right'
elif location=='B':
return 'left'

environment={
'A':'dirty',
'B':'dirty'
}

def environment_step(agent_location):
status=environment[agent_location]
action=reflex_agent(agent_location,status)
if action=='suck':
print('Agent cleaned',agent_location)
environment[agent_location]='clean'
elif action=='right':
print('Agent moved right from',agent_location)
agent_location='B'
elif action=='left':
print('Agent moved left from',agent_location)
agent_location='A'
return agent_location

def main():
agent_location='A'
while True:
agent_location=environment_step(agent_location)
if all(value=='clean' for value in environment.values()):
print('Environment is clean')
break
if __name__=='__main__':
main()
